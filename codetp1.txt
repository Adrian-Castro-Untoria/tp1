class Personne {
  final String nom;
  final String age;
  final String ville;

  Personne(this.nom, this.age, this.ville);

  static imprimerInfos(nom, age, ville) {
    return '${nom} is ${age} and lives at ${ville}';
  }

  @override
  String toString() {
    return '${nom} a ${age} et habite a ${ville}';
  }
}
